NP -> DetP Nbar
<NP> = hd-specifier-ph
<NP DTRS HEAD-DTR> = <Nbar>
<NP DTRS NON-HD-DTRS> = [<DetP>]

// p. 19 [2]
<NP DTRS HEAD-DTR PHON> = <NP DTRS HEAD-DTR DTRS HEAD-DTR PHON>

// p. 19 [4] SPEC/Specified
<NP DTRS NON-HD-DTRS FIRST HEAD-DTR SYNSEM LOC CAT HEAD SPEC> = <NP DTRS HEAD-DTR SYNSEM>

// p. 19 [5] SPR/Specifier
<NP DTRS HEAD-DTR SYNSEM LOC CAT SPR> = [<NP DTRS HEAD-DTR DTRS HEAD-DTR SYNSEM LOC CAT SPR>]
//<NP DTRS HEAD-DTR DTRS HEAD-DTR SYNSEM LOC CAT SPR> = [<NP DTRS HEAD-DTR `SYNSEM LOC CAT SPR>]
<NP DTRS HEAD-DTR SYNSEM LOC CAT SPR FIRST> = <NP DTRS NON-HD-DTRS FIRST SYNSEM>

DetP -> det
<DetP> = phrase
<DetP> = <det>

Nbar -> noun
<Nbar> = hd-complement-ph
<noun> = word
<Nbar DTRS HEAD-DTR PHON FIRST> = <noun PHON>
<Nbar DTRS HEAD-DTR SYNSEM LOC CAT HEAD> = <noun SYNSEM LOC CAT HEAD>
// SPR should be non-empty
<Nbar SYNSEM LOC CAT SPR> = [BOTTOM]
