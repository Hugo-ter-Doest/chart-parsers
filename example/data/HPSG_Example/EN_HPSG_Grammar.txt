NP -> DetP Nbar
<NP> = hd-specifier-ph

<NP DTRS HEAD-DTR> = <Nbar>
<NP DTRS NON-HD-DTRS FIRST> = <DetP>
<NP DTRS NON-HD-DTRS REST> = elist

// p. 19 [1] + [2] PHON
//<NP PHON FIRST> = <NP DTRS NON-HD-DTRS FIRST PHON>
//<NP PHON REST FIRST> = <NP DTRS HEAD-DTR PHON>
//<NP PHON REST REST> = BOTTOM

// p. 19 [3] Head Feature Principle
//<NP SYNSEM LOC CAT HEAD> = <NP DTRS HEAD-DTR SYNSEM LOC CAT HEAD>
//<NP DTRS HEAD-DTR SYNSEM LOC CAT HEAD> = <NP DTRS HEAD-DTR DTRS HEAD-DTR SYNSEM LOC CAT HEAD>

// p. 19 [2]
<NP DTRS HEAD-DTR PHON> = <NP DTRS HEAD-DTR DTRS HEAD-DTR PHON>

// p. 19 [4] SPEC/Specified
<NP DTRS NON-HD-DTRS FIRST HEAD-DTR SYNSEM LOC CAT HEAD SPEC> = <NP DTRS HEAD-DTR SYNSEM>

// p. 19 [5] SPR/Specifier
<NP DTRS HEAD-DTR SYNSEM LOC CAT SPR FIRST> = <NP DTRS HEAD-DTR DTRS HEAD-DTR SYNSEM LOC CAT SPR FIRST>
<NP DTRS HEAD-DTR SYNSEM LOC CAT SPR REST> = elist
<NP DTRS HEAD-DTR DTRS HEAD-DTR SYNSEM LOC CAT SPR REST> = elist
<NP DTRS HEAD-DTR SYNSEM LOC CAT SPR FIRST> = <NP DTRS NON-HD-DTRS FIRST SYNSEM>

// p.19 [6] Head Feature Principle
//<NP DTRS NON-HD-DTRS FIRST SYNSEM LOC CAT HEAD> = <NP DTRS NON-HD-DTRS FIRST HD-DTR SYNSEM LOC CAT HEAD>

DetP -> det
<DetP> = phrase
<DetP> = <det>

Nbar -> noun
<Nbar> = hd-complement-ph
<noun> = word
<Nbar DTRS HEAD-DTR PHON FIRST> = <noun PHON>
<Nbar DTRS HEAD-DTR SYNSEM LOC CAT HEAD> = <noun SYNSEM LOC CAT HEAD>
// SPR should be non-empty
<Nbar SYNSEM LOC CAT SPR> = list
<Nbar SYNSEM LOC CAT SPR FIRST> = synsem
<Nbar SYNSEM LOC CAT SPR REST> = elist
